--use left join to combine both tables

SELECT 
  A.UserID0,
  CASE WHEN B.Gender = 'None' THEN 'Unknown' ELSE B.Gender END as Gender,
  CASE WHEN B.Race = 'None' THEN 'Unknown' ELSE B.Race END as Race,
  COALESCE(B.Age, 0) as Age,
  CASE WHEN B.Province = 'None' THEN 'Unknown' ELSE B.Province END as Province,
  A.Channel2,
  from_utc_timestamp(A.RecordDate2, 'Africa/Johannesburg') as SA_DateTime,
  date_format(from_utc_timestamp(A.RecordDate2, 'Africa/Johannesburg'), 'yyyy-MM-dd') as SA_Date,
  date_format(from_utc_timestamp(A.RecordDate2, 'Africa/Johannesburg'), 'HH:mm:ss') as SA_Time,
  A.`Duration 2` as Duration,
  date_format(A.`Duration 2`, 'HH:mm:ss') as Duration_Formatted,
  A.userid4
FROM `workspace`.`default`.`bright_tv_v` A
LEFT JOIN `workspace`.`default`.`bright_tv_u` B
  ON A.UserID0 = B.UserID
LIMIT 100;

--TOTAL DURATION OF VIEWING BY CHANNEL (SA TIME)

SELECT 
  A.Channel2,
  COUNT(*) as total_views,
  SUM(
    HOUR(A.`Duration 2`) * 3600 + 
    MINUTE(A.`Duration 2`) * 60 + 
    SECOND(A.`Duration 2`)
  ) as total_duration_seconds,
  CONCAT(
    LPAD(CAST(FLOOR(SUM(HOUR(A.`Duration 2`) * 3600 + MINUTE(A.`Duration 2`) * 60 + SECOND(A.`Duration 2`)) / 3600) AS STRING), 2, '0'), ':',
    LPAD(CAST(FLOOR((SUM(HOUR(A.`Duration 2`) * 3600 + MINUTE(A.`Duration 2`) * 60 + SECOND(A.`Duration 2`)) % 3600) / 60) AS STRING), 2, '0'), ':',
    LPAD(CAST(FLOOR(SUM(HOUR(A.`Duration 2`) * 3600 + MINUTE(A.`Duration 2`) * 60 + SECOND(A.`Duration 2`)) % 60) AS STRING), 2, '0')
  ) as total_duration_formatted
FROM `workspace`.`default`.`bright_tv_v` A
LEFT JOIN `workspace`.`default`.`bright_tv_u` B
  ON A.UserID0 = B.UserID
GROUP BY A.Channel2
ORDER BY total_duration_seconds DESC;

----6. VIEWING BY AGE GROUP
SELECT
  CASE  
    WHEN B.Age <= 9 THEN 'Kids'
    WHEN B.Age BETWEEN 10 AND 16 THEN 'Teens'
    WHEN B.Age BETWEEN 17 AND 24 THEN 'Youth'
    WHEN B.Age BETWEEN 25 AND 35 THEN 'Adults'
    WHEN B.Age BETWEEN 36 AND 55 THEN 'Older Adults'
    WHEN B.Age BETWEEN 56 AND 75 THEN 'Seniors'
    WHEN B.Age BETWEEN 76 AND 99 THEN 'Elderly'
    ELSE 'Unknown'
  END AS Age_Group,
  CASE  
    WHEN B.Age <= 9 THEN '0-9'
    WHEN B.Age BETWEEN 10 AND 16 THEN '10-16'
    WHEN B.Age BETWEEN 17 AND 24 THEN '17-24'
    WHEN B.Age BETWEEN 25 AND 35 THEN '25-35'
    WHEN B.Age BETWEEN 36 AND 55 THEN '36-55'
    WHEN B.Age BETWEEN 56 AND 75 THEN '56-75'
    WHEN B.Age BETWEEN 76 AND 99 THEN '76-99'
    ELSE 'N/A'
  END AS Age_Bracket,
  COUNT(*) AS Count
FROM workspace.default.bright_tv_v A
JOIN workspace.default.bright_tv_u B
ON A.UserID0 = B.UserID
GROUP BY
  Age_Group,
  Age_Bracket
ORDER BY
  Count DESC;

--AGGREGATION: Daily Viewing Trends

SELECT 
  DATE(FROM_UTC_TIMESTAMP(RecordDate2, 'Africa/Johannesburg')) AS viewing_date_sa,
  DAYOFWEEK(FROM_UTC_TIMESTAMP(RecordDate2, 'Africa/Johannesburg')) AS day_of_week,
  CASE DAYOFWEEK(FROM_UTC_TIMESTAMP(RecordDate2, 'Africa/Johannesburg'))
    WHEN 1 THEN 'Sunday'
    WHEN 2 THEN 'Monday'
    WHEN 3 THEN 'Tuesday'
    WHEN 4 THEN 'Wednesday'
    WHEN 5 THEN 'Thursday'
    WHEN 6 THEN 'Friday'
    WHEN 7 THEN 'Saturday'
  END AS day_name,
  COUNT(*) AS total_views,
  COUNT(DISTINCT UserID) AS unique_viewers,
  SUM(HOUR(Duration_2) * 3600 + MINUTE(Duration_2) * 60 + SECOND(Duration_2)) AS total_seconds,
  CONCAT(
    LPAD(CAST(FLOOR(SUM(HOUR(Duration_2) * 3600 + MINUTE(Duration_2) * 60 + SECOND(Duration_2)) / 3600) AS STRING), 2, '0'), ':',
    LPAD(CAST(FLOOR((SUM(HOUR(Duration_2) * 3600 + MINUTE(Duration_2) * 60 + SECOND(Duration_2)) % 3600) / 60) AS STRING), 2, '0'), ':',
    LPAD(CAST(FLOOR(SUM(HOUR(Duration_2) * 3600 + MINUTE(Duration_2) * 60 + SECOND(Duration_2)) % 60) AS STRING), 2, '0')
  ) AS total_duration_formatted
FROM workspace.default.bright_tv_combined
WHERE RecordDate2 IS NOT NULL
GROUP BY DATE(FROM_UTC_TIMESTAMP(RecordDate2, 'Africa/Johannesburg')), DAYOFWEEK(FROM_UTC_TIMESTAMP(RecordDate2, 'Africa/Johannesburg'))
ORDER BY viewing_date_sa;
